\excalwd=\wd\excalbox

\parskip=0pt\offinterlineskip\parindent0pt
\hbox{\excala\char0\char1\char2\char3\char4}
\hbox{\excala\char5\char6\char7\excalb\char0\char1}

\hbox{\excalb\char7\excalc\char0\char1\char2\char3}}}
\ifx\parbox\undefined
    \def\setexcal{\box\excalbox}
\else
    \def\setexcal{\parbox{\wd\excalbox}{\box\excalbox}}
\fi
