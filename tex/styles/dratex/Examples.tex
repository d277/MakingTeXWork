@* The mapping.  The view
   {\tt (x,y)} is  on the
   {\tt Z=0} plane.
@<map(x,y) := mapping of
   (X,Y,Z)@>=
   x := X * D DIV (D + Z);

@  The viewer is located
   at {\tt (0,0,-D)}.
@<CONST + VAR @>=
   CONST D=100;
   VAR  X {x}, Z: INTEGER; 

@
@<map...@>=
   y := Y * D DIV (D + Z);

@
@<CON...@>=Y {y}: INTEGER;

@* Main program.
@p PROGRAM D2view; 
     @<CON...@>
   BEGIN  read(X,Y,Z);
     @<map...@>
     write(x,y)
   END.
>>>
