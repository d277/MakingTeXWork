\headrulewidth=0.25pt

\pagestyle{fancy}

\makeatletter  
\def\partmark#1{%
  \markboth{Part\space\thepart:\space#1}{Part\space\thepart:\space#1}%
  \gdef\@currentlabellevel{1}%
  \gdef\@currentlabeltext{#1}} % remember part title
\def\chaptermark#1{%
  \ifnum\c@chapter=0\relax%
    \markboth{#1}{#1}%
  \else%
    \markboth{\@chapapp\space\thechapter:\space#1}%
    {\@chapapp\space\thechapter:\space#1}%
  \fi%
  \gdef\@currentlabellevel{1}%
  \gdef\@currentlabeltext{#1}}% remember chapter title
\def\sectionmark#1{%
  \gdef\@currentlabellevel{2}%
  \gdef\@currentlabeltext{#1}} % forget about sections ...
\def\subsectionmark#1{%
  \gdef\@currentlabellevel{3}%
  \gdef\@currentlabeltext{#1}} % and subsections
\def\subsubsectionmark#1{%
  \gdef\@currentlabellevel{4}%
  \gdef\@currentlabeltext{#1}} % and subsubsections
\makeatother

\def\headfont{\rm\it\bf\size{9}{10pt}\selectfont}
